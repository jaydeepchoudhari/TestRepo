 Select f.id, f.field_label, f.answer_list_id, a.id, a.display_text From dbo.FIELD f
inner join dbo.SECTION s on s.id = f.section_id
inner join dbo.ANSWER_LIST al on al.id = f.answer_list_id
inner join dbo.ANSWER a on a.answers_list_id = al.id
 where template_id = 10044 and f.soft_deleted = 0
 and input_type = 10004 --and f.id = 19946
 order by f.field_order, f.id, a.display_order
